class Cars { constructor(brand, model, color, year) { this.brand = brand; this.model = model; this.color = color; this.year = year; } } class Mini extends Cars { constructor(brand, model, color, year, othername = "", bodyStyle = "") { super(brand, model, color, year); this.otherName = othername; this.bodyStyle = bodyStyle; } } class Pontiac extends Cars { constructor(brand, model, color, year, layout, platform) { super(brand, model, color, year); this.layout = layout; this.platform = platform; } }
